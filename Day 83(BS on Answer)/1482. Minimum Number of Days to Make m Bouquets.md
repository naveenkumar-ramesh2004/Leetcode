## [1482. Minimum Number of Days to Make m Bouquets](https://leetcode.com/problems/minimum-number-of-days-to-make-m-bouquets/description/)

```
class Solution {
    public int minDays(int[] bloomDay, int m, int k) {
        int low=Integer.MAX_VALUE;
        int high=-1;
        for(int val:bloomDay){
            low=Math.min(low,val);
            high=Math.max(high,val);
        }
        return BS(bloomDay,m,k,low,high);
    }
    int BS(int[] b,int m,int k,int low,int high){
        if(m*k>b.length){return -1;}
        int ans=-1;
        while(low<=high){
            int mid=(low+high)/2;
            if(canDo(b,m,k,mid)){
                ans=mid;
                high=mid-1;
            }
            else{
                low=mid+1;
            }
        }
        return ans;
    }
    boolean canDo(int[] b,int m,int k,int day){
        int kcount=0;
        int mcount=0;
        for(int val:b){
            if(val<=day){
                kcount++;
            }
            else{
                mcount+=kcount/k;
                kcount=0;
            }
            // if(val-day<=0){
            //     kcount++;
            //     if(kcount==k){
            //         mcount++;
            //         kcount=0;
            //     }
            // }
            // else{
            //     kcount=0;
            // }
        }
        mcount+=kcount/k;
        return mcount>=m;
    }
}
