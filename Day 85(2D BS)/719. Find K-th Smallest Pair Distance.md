## [719. Find K-th Smallest Pair Distance](https://leetcode.com/problems/find-k-th-smallest-pair-distance/description/)
```
class Solution {
    public int smallestDistancePair(int[] nums, int k) {
        Arrays.sort(nums);
        int high=nums[nums.length-1]-nums[0];
        return findsmalldis(nums,k,high);
    }
    int findsmalldis(int[] arr,int k,int high){
        int low=0;
        while(low<high){
            int mid = (low+high)/2;
            if(countDis(arr,k,mid)){
                high=mid;
            }
            else{
                low=mid+1;
            }
        }
        return low;
    }
    boolean countDis(int[] arr,int k,int dis){
        int count=0;
        int j=1;
        for(int i=0;i<arr.length;i++){
            while(j<arr.length && arr[j]-arr[i]<=dis){
                j++;
            }
            count+=(j-i-1);
        }
        System.out.println(count+" "+dis);
        return count>=k;
    }
}
