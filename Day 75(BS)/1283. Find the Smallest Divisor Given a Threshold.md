## [1283. Find the Smallest Divisor Given a Threshold](https://leetcode.com/problems/find-the-smallest-divisor-given-a-threshold/description/)


```
class Solution {
    public int smallestDivisor(int[] nums, int threshold) {
        int max=-1;
        for(int val:nums){
            max=Math.max(max,val);
        }
        return findSmallestDivisor(nums,threshold,max);
    }
    int findSmallestDivisor(int[] nums,int t,int end){
        int start=1;
        int ans=-1;
        while(start<=end){
            int mid = start+(end-start)/2;
            int sum=0;
            for(int value:nums){
                if(value%mid==0){
                    sum+=(value/mid);
                }
                else{
                    sum+=((value/mid)+1);
                }
            if(sum>t){
                start=mid+1;
                continue;
            }
            }
            if(sum<=t){
                ans=mid;
                end=mid-1;
            }
        }
        return ans;
    }


}
