class Solution {
    public int compress(char[] chars) {
        return groupandCount(chars);
    }
    int groupandCount(char[] chars){
        int count = 1;
        int idx=0;
        for(int i=0;i<chars.length;i++){  // find the group and count 
            if(i+1<chars.length && chars[i]==chars[i+1]){
                count++;
            }
            else{
                chars[idx++]=chars[i]; //add No Matter the count
                if(count>=2){ //Here I learn How to inser a (int)-> 12 to '1','2' use 12->(string.valueOf) "12" -> toCharArray()-> '1','2';
                    for(char c:String.valueOf(count).toCharArray()){
                    chars[idx++]=c;
                    }
                }
                count=1;
            }
        }
        return idx;
    }
}
