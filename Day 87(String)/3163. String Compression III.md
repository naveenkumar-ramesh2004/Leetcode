## [3163. String Compression III](https://leetcode.com/problems/string-compression-iii/description/)

```
class Solution {
    public String compressedString(String word) {
        return groupandCount(word);
    }
    String groupandCount(String word){
        int count = 1;
        StringBuilder comp = new StringBuilder();
        for(int i=0;i<word.length();i++){  // find the group and count 
            if(i<word.length()-1 && word.charAt(i)==word.charAt(i+1)){
                count++;
            }
            else{
                if(count>9){ // count > 9
                    for(int j=0;j<count/9;j++){ // how many 9 there add all example 19 -> 2(9) there so add
                        comp.append('9');
                        comp.append(word.charAt(i));
                    }
                    int remain=count-(count/9)*9; // chaeck remaining
                    if(remain>0){
                        comp.append(remain);
                        comp.append(word.charAt(i));
                    }
                }
                else{ 
                    comp.append(count);
                    comp.append(word.charAt(i));
                }
                count=1;
            }
        }
        return comp.toString();
    }
}
