    String countAndSay(String word){
        int count = 1;
        StringBuilder comp = new StringBuilder();
        for(int i=0;i<word.length();i++){  // I scan the string from left to right and group consecutive characters.
            if(i<word.length()-1 && word.charAt(i)==word.charAt(i+1)){
                count++;
            }
            else{
                comp.append(count);
                comp.append(word.charAt(i));
                count=1;
            }
        }
        return comp.toString();
    }
}
