## [128. Longest Consecutive Sequence](https://leetcode.com/problems/longest-consecutive-sequence/description/)

```
class Solution {
    public int longestConsecutive(int[] nums) {
        if(nums.length==0){return 0;}
        Set<Integer> arr= new HashSet<>();

        // Add value in HashMap 

        for(int val:nums){
            arr.add(val);
        }

        int count=1,max=1;

        for(int val:arr){
            // Find the min value [4,3] chech (4-1) =3 present are not if three present skip 4 then 3-1 =2 not there so 2 is min
            if(!arr.contains(val-1)){  // find the min value in consequtive
                count=1;
                //find all consequtive max value
                while(arr.contains(++val)){
                    count++; 
                }
                max=Math.max(max,count);
            }
            
        }
        return max;
    }
}
