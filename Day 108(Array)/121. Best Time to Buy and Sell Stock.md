class Solution {
    public int maxProfit(int[] prices) {
        int min=Integer.MAX_VALUE;
        int profit=0;
        for(int val:prices){
            if(min>val){
                // min price should be buy
                min=val;
            }
            else{
                // max price should be sell
                if(val-min>profit){
                    profit=val-min;
                }
            }
        }
        return profit;
    }
}
