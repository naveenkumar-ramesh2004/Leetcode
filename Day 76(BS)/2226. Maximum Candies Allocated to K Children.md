## [2226. Maximum Candies Allocated to K Children](https://leetcode.com/problems/maximum-candies-allocated-to-k-children/?source=submission-ac)

```
class Solution {
    public int maximumCandies(int[] candies, long k) {
     int max=-1;
        for(int val:candies){
            max=Math.max(max,val);
        }
        return findMaxCandies(candies,k,max);
    }
    int findMaxCandies(int[] nums,long k,int end){
        int start=1;
        int ans=0;
        while(start<=end){
            int mid = start+(end-start)/2;
            long count=0;
            for(int value:nums){
                int p=value/mid;
                if(p>0){
                    count+=p;
                }
            }
            if(count>=k){
                start=mid+1;
                ans=mid;
            }
            if(count<k){
                end=mid-1;
            }
        }
        return ans;
    }
}
