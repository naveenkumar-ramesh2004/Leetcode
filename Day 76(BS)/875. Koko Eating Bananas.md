## [875. Koko Eating Bananas](https://leetcode.com/problems/koko-eating-bananas/description/)

```
class Solution {
    public int minEatingSpeed(int[] piles, int h) {
        int max=-1;
        for(int val:piles){
            max=Math.max(max,val);
        }
        return findMinBanaEat(piles,h,max);
    }
    int findMinBanaEat(int[] nums,int t,int end){
        int start=1;
        int ans=-1;
        while(start<=end){
            int mid = start+(end-start)/2;
            int sum=0;
            for(int value:nums){
                if(value%mid==0){
                    sum+=(value/mid);
                }
                else{
                    sum+=((value/mid)+1);
                }
            if(sum>t){
                start=mid+1;
                break;
            }
            }
            if(sum<=t){
                ans=mid;
                end=mid-1;
            }
        }
        return ans;
    }
}
