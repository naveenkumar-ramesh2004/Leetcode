## [20. Valid Parentheses](https://leetcode.com/problems/valid-parentheses/description/)



#### 3Stack
```
class Solution {
    public boolean isValid(String s) {
        Stack<Character> container = new Stack<>();
        for(char c:s.toCharArray()){
            if(c=='('||c=='['||c=='{'){
                container.push(c);
            }
            else{
                if(container.size()==0){return false;}
                switch(c){
                    case ')':
                    if(container.pop()!='('){
                        return false;
                    }
                    break;
                    case ']':
                    if(container.pop()!='['){
                        return false;
                    }
                    break;
                    default:
                    if(container.pop()!='{'){
                        return false;
                    }
                    break;
                }
            }
        }
        return (container.size()!=0)? false:true;
    }
}

class Solution {
    public static boolean isValid(String s) {
        char[] arr = new char[s.length()];
        int i = 0; 
        for (char ch : s.toCharArray()) {
            if (ch == '(' || ch == '{' || ch == '[') {
                arr[i++] = ch;
            } else {
                if (i == 0) return false;

                char last = arr[i - 1];
                if ((ch == ')' && last == '(') ||
                    (ch == '}' && last == '{') ||
                    (ch == ']' && last == '[')) {
                    i--; 
                } else {
                    return false; 
                }
            }
        }
        return i == 0;
    }
