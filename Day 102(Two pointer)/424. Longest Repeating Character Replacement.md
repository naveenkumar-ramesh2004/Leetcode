## [424. Longest Repeating Character Replacement](https://leetcode.com/problems/longest-repeating-character-replacement/description/)
```
class Solution {
    public int characterReplacement(String s, int k) {
        
    // return bruteforce(s,k);
    // return better(s,k);
     return optimal(s,k);
    }
    int bruteforce(String s,int k){
        int max=0;
        for(int i=0;i<s.length();i++){
            char c=s.charAt(i); // take a character 
            for(int x=0;x<s.length();x++){
            int convert=0; //count how many convert 
            int j=x;
            while(j<s.length() && convert<=k){ // convert exit
                if(c!=s.charAt(j++)){
                    convert++;
                }
            }
            if(convert>k){j--;} // convert exit reduce -1;
            max=Math.max(max,j-x); // find max length
            }
        }
        return max;
    }
    int better(String s,int k){
        int max=0;
        for(int i=0;i<s.length();i++){
            int maxfreq=0;
            int j=i;
            HashMap<Character,Integer> freq = new HashMap<>();
            while(j<s.length()&&(j-i)-maxfreq<=k){
                char c=s.charAt(j);
                freq.put(c,freq.getOrDefault(c,0)+1);
                maxfreq=Math.max(freq.get(c),maxfreq);
                j++;
            }
            if((j-i)-maxfreq>k)j--;
            max=Math.max(j-i,max);
        }
        return max;
    }
    int optimal(String s,int k){
        int l=0;
        int r=0;
        int maxfreq=0;
        int ans=0;
        int[] freq = new int[26];
        while(r<s.length()){
            freq[s.charAt(r)-'A']++;
            maxfreq=Math.max(maxfreq,freq[s.charAt(r)-'A']);
            if((r-l+1)-maxfreq>k){
                freq[s.charAt(l)-'A']--;
                maxfreq=0;
                // for(int c:freq)maxfreq=Math.max(maxfreq,c);
                l++;
            }
            ans=Math.max(ans,(r-l+1));
            r++;
        }
        return ans;
    }
}
